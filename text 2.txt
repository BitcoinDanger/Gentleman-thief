const $ = id => document.getElementById(id);
function showScreen(id){
  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));
  $(id).classList.add('active');
  if(id==='mapscreen') renderMap();
  if(id==='upgradescreen') renderUpgrades();
}
function toast(msg){
  const t=$('toast');
  t.textContent=msg;
  t.classList.add('show');
  clearTimeout(t._t);
  t._t=setTimeout(()=>t.classList.remove('show'),2800);
}
function fmt(n){
  if(n>=1000000000) return '$'+(n/1000000000).toFixed(0)+'B';
  if(n>=1000000)    return '$'+(n/1000000).toFixed(1)+'M';
  if(n>=1000)       return '$'+(n/1000).toFixed(0)+'K';
  return '$'+n;
}

// ============================================================
//  MAP RENDER
// ============================================================
const PHASES = ['PHASE I — CITY TARGETS','PHASE II — CORPORATE HEISTS','PHASE III — INTERNATIONAL OPS','PHASE IV — GOVERNMENT & MILITARY','PHASE V — LEGENDARY HEISTS'];
function renderMap(){
  $('m-rep').textContent  = G.rep;
  $('m-loot').textContent = fmt(G.totalLoot);
  $('m-wins').textContent = G.wins+'/15';
  $('m-pts').textContent  = G.upgPts;
  const scroll = $('map-scroll');
  scroll.innerHTML='';
  PHASES.forEach(phase=>{
    const ms = MISSIONS.filter(m=>m.phase===phase);
    const lbl=document.createElement('div');
    lbl.className='map-phase-label';
    lbl.textContent=phase;
    scroll.appendChild(lbl);
    const grid=document.createElement('div');
    grid.className='map-grid';
    ms.forEach(m=>{
      const isLocked = m.id>0 && !G.done.includes(m.id-1);
      const isDone   = G.done.includes(m.id);
      const node=document.createElement('div');
      node.className=`map-node ${m.phaseClass}${isLocked?' locked':''}${isDone?' done':''}`;
      node.innerHTML=`
        <div class="map-node-emoji">${m.emoji}</div>
        <div class="map-node-lvl">LVL ${m.level}</div>
        <div class="map-node-name">${m.name}</div>
        <div class="map-node-loot">${fmt(m.loot)}</div>
        ${isLocked?'<div class="map-lock-icon">🔒</div>':''}
      `;
      if(!isLocked) node.onclick=()=>startMission(m.id);
      grid.appendChild(node);
    });
    scroll.appendChild(grid);
  });
}

// ============================================================
//  UPGRADES RENDER
// ============================================================
function renderUpgrades(){
  $('upg-pts-display').textContent = G.upgPts;
  ['stealth','combat','gadget'].forEach(cat=>{
    const grid = $('upg-'+cat+'-grid');
    grid.innerHTML='';
    UPGRADES[cat].forEach(u=>{
      const lv=G.upgrades[u.key];
      const maxed=lv>=u.maxLv;
      const canAfford=G.upgPts>=u.cost;
      const pips=Array.from({length:u.maxLv},(_,i)=>`<div class="upg-pip${i<lv?' filled':''}"></div>`).join('');
      const card=document.createElement('div');
      card.className='upg-card'+(maxed?' maxed':'');
      card.innerHTML=`
        <div class="upg-card-icon">${u.icon}</div>
        <div class="upg-card-name">${u.name}</div>
        <div class="upg-card-desc">${u.desc}</div>
        <div class="upg-card-level">${pips}</div>
        <button class="upg-card-btn${maxed?' maxed-btn':''}" ${(!canAfford&&!maxed)?'disabled':''} onclick="buyUpgrade('${u.key}','${cat}')">
          ${maxed?'✓ MAXED':`UPGRADE (${u.cost} PTS)`}
        </button>
      `;
      grid.appendChild(card);
    });
  });
}
function buyUpgrade(key,cat){
  const u = UPGRADES[cat].find(x=>x.key===key);
  if(!u||G.upgrades[key]>=u.maxLv||G.upgPts<u.cost) return;
  G.upgrades[key]++;
  G.upgPts-=u.cost;
  toast(`✅ ${u.name} upgraded to Lv ${G.upgrades[key]}`);
  renderUpgrades();
}

function startMission(id){
  const m=MISSIONS[id];
  G.current=id; G.step=0;
  G.hp=100; G.stealth=100; G.suspicion=0;
  G.runLoot=m.loot; G.runRep=m.rep;
  $('scene-emoji').textContent=m.emoji;
  $('scene-loc').textContent=m.loc;
  $('scene-phase').textContent=m.phase;
  updateHUD();
  showScreen('game');
  renderStep(0);
}

function renderStep(idx){
  const m=MISSIONS[G.current];
  const s=m.steps[idx];
  G.step=idx;
  $('n-speaker').textContent=s.sp;
  $('n-text').textContent=s.txt;
  $('combat-modal').classList.remove('active');
  const box=$('choices');
  box.innerHTML='';
  s.ch.forEach(c=>{
    const btn=document.createElement('button');
    btn.className='choice-btn';
    btn.innerHTML=`<span class="choice-icon">${c.i}</span><span class="choice-text">${c.t}</span><span class="choice-tag ${c.tc}">${c.tl}</span>`;
    btn.onclick=()=>pick(c);
    box.appendChild(btn);
  });
}

// ============================================================
//  PICK CHOICE — apply upgrades
// ============================================================
function pick(c){
  // Apply upgrade modifiers
  let susReduction = 1 - (G.upgrades.shadow * 0.15);

  if(c.ef==='st'){
    if(c.ev>0){
      let bonus = c.ev + (G.upgrades.lockpick*10*(c.i==='🔑'||c.t.toLowerCase().includes('lock')?1:0));
      G.stealth=Math.min(100,G.stealth+bonus);
      toast(`🤫 STEALTH +${c.ev}`);
    } else {
      G.stealth=Math.max(0,G.stealth+c.ev);
      let sus = Math.round(-c.ev * susReduction);
      G.suspicion=Math.min(100,G.suspicion+sus);
      toast(`⚠ SUSPICION RISING`);
    }
  } else if(c.ef==='sus'){
    let base = Math.round(c.ev * susReduction);
    if(c.t.toLowerCase().includes('hack')||c.i==='💻') base=Math.max(0,base-(G.upgrades.hacker*10));
    G.suspicion=Math.min(100,G.suspicion+base);
    if(base>0) toast(`⚠ SUSPICION +${base}`);
    else toast(`✅ SUSPICION NEUTRALISED`);
  } else if(c.ef==='loot'){
    G.runLoot+=c.ev; toast(`💰 BONUS LOOT +${fmt(c.ev)}`);
  } else if(c.ef==='combat'){
    updateHUD();
    triggerFight(()=>{ c.nx==='win'?endMission(true):renderStep(c.nx); });
    return;
  }
  updateHUD();
  if(G.hp<=0){ endMission(false); return; }
  if(c.nx==='win') endMission(true);
  else renderStep(c.nx);
  // random guard event
  if(G.suspicion>=65 && Math.random()<(0.3-(G.upgrades.scanner*0.08))){
    setTimeout(()=>{ toast('💂 UNSCHEDULED PATROL!'); triggerFight(()=>{}); },800);
  }
}

function updateHUD(){
  $('bar-hp').style.width  = G.hp+'%';
  $('bar-st').style.width  = G.stealth+'%';
  $('bar-sus').style.width = G.suspicion+'%';
  const a=$('alert');
  if(G.suspicion<30)      { a.textContent='● SAFE';    a.className='alert-pill safe'; }
  else if(G.suspicion<65) { a.textContent='● CAUTION'; a.className='alert-pill caution'; }
  else                    { a.textContent='● DANGER';  a.className='alert-pill danger'; }
}

// ============================================================
//  COMBAT
// ============================================================
function triggerFight(cb){
  G.fightCb=cb;
  $('combat-modal').classList.add('active');
  $('combat-msg').textContent='A guard has spotted Jackal. Choose your move:';
  $('combat-enemy').textContent='💂';
  $('combat-enemy2').textContent='💂';
}
function fight(act){
  const msg=$('combat-msg'), en=$('combat-enemy'), en2=$('combat-enemy2');
  let ok=false;
  const resist = G.upgrades.resilience * 5;
  if(act==='shoot'){
    let rate = 0.75 + (G.upgrades.accuracy*0.08);
    ok=Math.random()<Math.min(0.95,rate);
    if(ok){ G.suspicion=Math.min(100,G.suspicion+22); msg.textContent='🔫 Shot fired. Guard down — but noise carries. Suspicion +22.'; }
    else  { G.hp=Math.max(0,G.hp-Math.max(5,20-resist)); msg.textContent=`🔫 He fired first. Jackal takes a hit. HP -${20-resist}.`; }
  } else if(act==='evade'){
    let base = G.upgrades.evasion>=2 ? 0.75 : G.stealth/110;
    ok=Math.random()<Math.min(0.9,base+(G.upgrades.evasion*0.1));
    if(ok){ G.stealth=Math.max(0,G.stealth-8); msg.textContent='🏃 Slipped into shadow. Close. Stealth -8.'; }
    else  { G.hp=Math.max(0,G.hp-Math.max(5,15-resist)); G.suspicion=Math.min(100,G.suspicion+15); msg.textContent=`🏃 Evade failed. Took a blow. HP -${15-resist}, Suspicion +15.`; }
  } else if(act==='disarm'){
    let rate = 0.6 + (G.upgrades.disarmSkill*0.12);
    ok=Math.random()<Math.min(0.95,rate);
    if(ok){ G.suspicion=Math.max(0,G.suspicion-5); msg.textContent='🥋 Textbook disarm. Guard down, silent and unhurt. Pure Jackal.'; }
    else  { G.hp=Math.max(0,G.hp-Math.max(5,25-resist)); msg.textContent=`🥋 Overconfident. HP -${25-resist}.`; }
  } else {
    const fleeAlwaysWorks = G.upgrades.smokeKit>=3;
    ok= fleeAlwaysWorks || Math.random()<0.85;
    const stCost = Math.max(5, 18-(G.upgrades.smokeKit*5));
    G.stealth=Math.max(0,G.stealth-stCost); G.suspicion=Math.min(100,G.suspicion+6);
    msg.textContent=`💨 Fled. He'll remember the red blazer. Stealth -${stCost}.`;
  }
  updateHUD();
  en.textContent=ok?'😵':'😡';
  en2.textContent=ok?'😵':'😡';
  setTimeout(()=>{
    $('combat-modal').classList.remove('active');
    if(G.hp<=0) endMission(false);
    else if(G.fightCb){ G.fightCb(); G.fightCb=null; }
  },2200);
}

// ============================================================
//  END MISSION
// ============================================================
function endMission(win){
  const ranks=['F','C','B','A','S','S+'];
  let ri=0;
  if(win){
    if(G.stealth>80&&G.suspicion<12) ri=5;
    else if(G.stealth>65&&G.suspicion<25) ri=4;
    else if(G.stealth>50&&G.suspicion<45) ri=3;
    else if(G.stealth>35) ri=2;
    else if(G.hp>50) ri=1;
    G.wins++; G.totalLoot+=G.runLoot; G.rep+=G.runRep;
    const pts = MISSIONS[G.current].pts;
    G.upgPts += pts;
    if(G.suspicion<20) G.ghosts++;
    if(!G.done.includes(G.current)) G.done.push(G.current);
    if(G.current+1<MISSIONS.length) MISSIONS[G.current+1].locked=false;
    $('o-title').textContent = G.suspicion<12?'👻 GHOST COMPLETE':'✅ HEIST COMPLETE';
    $('o-title').className='outcome-title win';
    $('o-quote').textContent = G.suspicion<12
      ? '"They found nothing. Not a shadow, not a footprint."'
      : '"Another city, another legend. The Jackal never sleeps."';
    $('o-pts').textContent='+'+pts;
    $('o-btn').textContent='NEXT MISSION';
    $('o-btn').onclick=()=>showScreen('mapscreen');
  } else {
    $('o-title').textContent='💀 MISSION FAILED';
    $('o-title').className='outcome-title lose';
    $('o-quote').textContent='"Even the finest blade can dull. Rest. Recalibrate. Return."';
    $('o-pts').textContent='+0';
    $('o-btn').textContent='TRY AGAIN';
    $('o-btn').onclick=()=>startMission(G.current);
    G.runLoot=0; G.runRep=0; ri=0;
  }
  $('o-loot').textContent=fmt(G.runLoot);
  $('o-rep').textContent ='+'+G.runRep;
  $('o-rank').textContent=win?ranks[ri]:'F';
  showScreen('outcome');
}
